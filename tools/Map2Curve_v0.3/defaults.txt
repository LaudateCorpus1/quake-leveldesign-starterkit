


// default values for drag&drop map-files

rad	0	// arc radius [0 = original radius (y-coord); game units = custom radius]
offset	0	// additional offset on top of radius [game units]
res	0	// number of arc sides [0 = auto resolution based on previous arc]
type	0	// type of construction frame [0 = Pi Circle; 1 = Grid Circle; 2 = Grid Path]

shift	0	// horizontal texture shift mode [0=Nullshift; 1=Section; 2=Brush; 3=Brush Texture; 4=Aligned Left(Section)]

height	0	// adds custom height to every section (stair steps) [game units]
ramp	0	// uses ramp mode for height setting (combined with height) [0=Off; 1=Linear; 2=Smooth]
ramptex 0	// defines texture alignment for ramp generation [0=sheared; 1=sloped] (0 is equal to JACK's UV Lock)

round	0	// rounds all coordinates to integer numbers [0,1]
tri	0	// triangulation for 5 and 6 sided brushes with 1 or 2 upright front/back faces [0,1]

obj	0	// exports individual arcs as OBJ files [0,1]

p_cornerfix 0	// re-arranges inner corners when "type" is 2 (Grid Path), to avoid overlapping meshes [0,1]
p_reverse  0	// reverses path direction (forwards<>backwards) [0,1]
p_split	0	// whether or not a final path extrusion object is being split into smaller parts on file export [0,1]
